// New function
	public void EncryptUtility() {
		string encryptPath = Custom.RootFolderDirectory() + "Encrypt";
		if (Directory.Exists(encryptPath)) {
			foreach (InGameTranslator.LanguageID language in Enum.GetValues(typeof(InGameTranslator.LanguageID))) {
				string langDir = "Text_" + LocalizationTranslator.LangShort(language);
				if (!Directory.Exists(encryptPath + "/" + langDir))
					Directory.CreateDirectory(encryptPath + "/" + langDir);
				FileInfo[] files = new DirectoryInfo(encryptPath).GetFiles();
				for (int i = 0; i < files.Length; i++) {
					if (files[i].Name.Substring(files[i].Name.Length - 4, 4) == ".txt") {
						int num = int.Parse(files[i].Name.Substring(0, files[i].Name.Length - 4));
						string text = File.ReadAllText(encryptPath + "/" + files[i].Name, Encoding.Default);
						if (text[0] == '0') {
							string text2 = Custom.xorEncrypt(text, (int)(54 + num + (int)language * 7));
							text2 = "1" + text2.Remove(0, 1);
							File.WriteAllText(encryptPath + "/" + langDir + "/" + files[i].Name, text2);
						}
					}
				}
			}
		}
	}
